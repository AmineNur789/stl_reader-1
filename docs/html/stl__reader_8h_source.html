176  // a coordinate triple with an additional index. The index is required
177  // for RemoveDoubles, so that triangles can be reindexed properly.
178  template <typename number_t, typename index_t>
179  struct CoordWithIndex {
180  number_t data[3];
181  index_t index;
182 
183  bool operator == (const CoordWithIndex& c) const
184  {
185  return c[0] == data[0] && c[1] == data[1] && c[2] == data[2];
186  }
187 
188  bool operator != (const CoordWithIndex& c) const
189  {
190  return c[0] != data[0] || c[1] != data[1] || c[2] != data[2];
191  }
192 
193  bool operator < (const CoordWithIndex& c) const
194  {
195  return data[0] < c[0]
196  || data[0] == c[0] && data[1] < c[1]
197  || data[0] == c[0] && data[1] == c[1] && data[2] < c[2];
198  }
199 
200  inline number_t& operator [] (const size_t i) {return data[i];}
201  inline number_t operator [] (const size_t i) const {return data[i];}
202  };

204  // sorts the array coordsWithIndexInOut and copies unique indices to coordsOut.
205  // Triangle-corners are re-indexed on the fly
206  template <class TNumberContainer, class TIndexContainer>
207  void RemoveDoubles (TNumberContainer& uniqueCoordsOut,
208  TIndexContainer& trisInOut,
209  std::vector <CoordWithIndex<
210  typename TNumberContainer::value_type,
211  typename TIndexContainer::value_type> >
212  &coordsWithIndexInOut)
213  {
214  using namespace std;
215 
216  typedef typename TNumberContainer::value_type number_t;
217  typedef typename TIndexContainer::value_type index_t;
218 
219  sort (coordsWithIndexInOut.begin(), coordsWithIndexInOut.end());
220 
221  // first count unique indices
222  size_t numUnique = 1;
223  for(size_t i = 1; i < coordsWithIndexInOut.size(); ++i){
224  if(coordsWithIndexInOut[i] != coordsWithIndexInOut[i - 1])
225  ++numUnique;
226  }
227 
228  uniqueCoordsOut.resize (numUnique * 3);
229  vector<index_t> newIndex (coordsWithIndexInOut.size());
230 
231  // copy unique coordinates to 'uniqueCoordsOut' and create an index-map
232  // 'newIndex', which allows to re-index triangles later on.
233  size_t curInd = 0;
234  newIndex[0] = 0;
235  for(size_t i = 0; i < 3; ++i)
236  uniqueCoordsOut[i] = coordsWithIndexInOut[0][i];
237 
238  for(size_t i = 1; i < coordsWithIndexInOut.size(); ++i){
239  const CoordWithIndex <number_t, index_t> c = coordsWithIndexInOut[i];
240  if(c != coordsWithIndexInOut[i - 1]){
241  ++curInd;
242  for(size_t j = 0; j < 3; ++j)
243  uniqueCoordsOut[curInd * 3 + j] = coordsWithIndexInOut[i][j];
244  }
245 
246  newIndex[c.index] = curInd;
247  }
248 
249  // re-index triangles, so that they refer to 'uniqueCoordsOut'
250  for(size_t i = 0; i < trisInOut.size(); ++i){
251  trisInOut[i] = newIndex[trisInOut[i]];
252  }
253  }
254 }// end of namespace stl_reader_impl

271 template <class TNumberContainer, class TIndexContainer>
272 bool ReadStlFile_ASCII(const char* filename,
273  TNumberContainer& coordsOut,
274  TNumberContainer& normalsOut,
275  TIndexContainer& trisOut,
276  TIndexContainer& solidRangesOut)
277 {
278  using namespace std;
279  using namespace stl_reader_impl;
280 
281  typedef typename TNumberContainer::value_type number_t;
282  typedef typename TIndexContainer::value_type index_t;
283 
284  coordsOut.clear();
285  normalsOut.clear();
286  trisOut.clear();
287  solidRangesOut.clear();
288 
289  ifstream in(filename);
290  READ_STL_COND_THROW(!in, "Couldnt open file " << filename);
291 
292  vector<CoordWithIndex <number_t, index_t>> coordsWithIndex;
293 
294  string buffer;
295  vector<string> tokens;
296  int lineCount = 1;
297  int solidIndex = -1;
298  int maxNumTokens = 0;
299  size_t numFaceVrts = 0;
300 
301  while(!(in.eof() || in.fail()))
302  {
303  // read the line and tokenize
304  getline(in, buffer);
305 
306  tokens.resize(maxNumTokens);
307  istringstream line(buffer);
308  int tokenCount = 0;
309  while(!(line.eof() || line.fail())){
310  if(tokenCount >= maxNumTokens){
311  maxNumTokens = tokenCount + 1;
312  tokens.resize(maxNumTokens);
313  }
314  line >> tokens[tokenCount];
315  ++tokenCount;
316  }
317 
318  tokens.resize(tokenCount);
319 
320  if(!tokens.empty())
321  {
322  string& tok = tokens[0];
323  if(tok.compare("vertex") == 0){
324  if(tokens.size() < 4){
325  READ_STL_THROW("ERROR while reading from " << filename <<
326  ": vertex not specified correctly in line " << lineCount);
327  }
328 
329  // read the position
330  CoordWithIndex <number_t, index_t> c;
331  for(size_t i = 0; i < 3; ++i)
332  c[i] = atof(tokens[i+1].c_str());
333  c.index = static_cast<index_t>(coordsWithIndex.size());
334  coordsWithIndex.push_back(c);
335  ++numFaceVrts;
336  }
337  else if(tok.compare("facet") == 0)
338  {
339  READ_STL_COND_THROW(tokens.size() < 5,
340  "ERROR while reading from " << filename <<
341  ": triangle not specified correctly in line " << lineCount);
342 
343  READ_STL_COND_THROW(tokens[1].compare("normal") != 0,
344  "ERROR while reading from " << filename <<
345  ": Missing normal specifier in line " << lineCount);
346 
347  // read the normal
348  for(size_t i = 0; i < 3; ++i)
349  normalsOut.push_back (atof(tokens[i+2].c_str()));
350 
351  numFaceVrts = 0;
352  }
353  else if(tok.compare("outer") == 0){
354  READ_STL_COND_THROW ((tokens.size() < 2) || (tokens[1].compare("loop") != 0),
355  "ERROR while reading from " << filename <<
356  ": expecting outer loop in line " << lineCount);
357  }
358  else if(tok.compare("endfacet") == 0){
359  READ_STL_COND_THROW(numFaceVrts != 3,
360  "ERROR while reading from " << filename <<
361  ": bad number of vertices specified for face in line " << lineCount);
362 
363  trisOut.push_back(coordsWithIndex.size() - 3);
364  trisOut.push_back(coordsWithIndex.size() - 2);
365  trisOut.push_back(coordsWithIndex.size() - 1);
366  }
367  else if(tok.compare("solid") == 0){
368  solidRangesOut.push_back(trisOut.size() / 3);
369  }
370  }
371  lineCount++;
372  }
373 
374  solidRangesOut.push_back(trisOut.size() / 3);
375 
376  RemoveDoubles (coordsOut, trisOut, coordsWithIndex);
377 
378  return true;
379 }

382 template <class TNumberContainer, class TIndexContainer>
383 bool ReadStlFile_BINARY(const char* filename,
384  TNumberContainer& coordsOut,
385  TNumberContainer& normalsOut,
386  TIndexContainer& trisOut,
387  TIndexContainer& solidRangesOut)
388 {
389  using namespace std;
390  using namespace stl_reader_impl;
391 
392  typedef typename TNumberContainer::value_type number_t;
393  typedef typename TIndexContainer::value_type index_t;
394 
395  coordsOut.clear();
396  normalsOut.clear();
397  trisOut.clear();
398  solidRangesOut.clear();
399 
400  ifstream in(filename, ios::binary);
401  READ_STL_COND_THROW(!in, "Couldnt open file " << filename);
402 
403  char stl_header[80];
404  in.read(stl_header, 80);
405  READ_STL_COND_THROW(!in, "Error while parsing binary stl header in file " << filename);
406 
407  uint32 numTris = 0;
408  in.read((char*)&numTris, 4);
409  READ_STL_COND_THROW(!in, "Couldnt determine number of triangles in binary stl file " << filename);
410 
411  vector<CoordWithIndex <number_t, index_t>> coordsWithIndex;
412 
413  for(uint32 tri = 0; tri < numTris; ++tri){
414  float d[12];
415  in.read((char*)d, 12 * 4);
416  READ_STL_COND_THROW(!in, "Error while parsing trianlge in binary stl file " << filename);
417 
418  for(int i = 0; i < 3; ++i)
419  normalsOut.push_back (d[i]);
420 
421  for(size_t ivrt = 1; ivrt < 4; ++ivrt){
422  CoordWithIndex <number_t, index_t> c;
423  for(size_t i = 0; i < 3; ++i)
424  c[i] = d[ivrt * 3 + i];
425  c.index = static_cast<index_t>(coordsWithIndex.size());
426  coordsWithIndex.push_back(c);
427  }
428 
429  trisOut.push_back(coordsWithIndex.size() - 3);
430  trisOut.push_back(coordsWithIndex.size() - 2);
431  trisOut.push_back(coordsWithIndex.size() - 1);
432 
433  char addData[2];
434  in.read(addData, 2);
435  READ_STL_COND_THROW(!in, "Error while parsing additional triangle data in binary stl file " << filename);
436  }
437 
438  solidRangesOut.push_back(0);
439  solidRangesOut.push_back(trisOut.size() / 3);
440 
441  RemoveDoubles (coordsOut, trisOut, coordsWithIndex);
442 
443  return true;
444 }
